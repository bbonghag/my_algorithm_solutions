# 1. 병렬 처리와 인덱싱의 효율화
## 1.1 zip() 함수를 활용한 데이터 결합 및 매핑
- 복수의 시퀀스 자료형을 동시에 순회할 때, 전통적 방식인 인덱스 접근은 코드가 복잡해지고 실수할 가능성을 높인다
- zip() 함수는 여러 반복 가능 객체 (Iterable)를 병렬로 묶어주는 효율적인 도구이다

- 동작 원리 : 각 Iterable에서 동일한 순서의 요소를 추출하여 튜플로 묶어주는 Iterator를 생성
- 주의 사항 : 인자로 전달된 리스트들의 길이가 다를 경우, 가장 짧은 리스트를 기준으로 연산이 종료됨 
    - Python 3.10 이상에서는 strict=True 옵션을 통해 길이가 다를 경우 예외를 발생시킬 수 있음

```
# [실전 예시] 두 리스트를 활용한 딕셔너리 일괄 생성
subjects = ['Math', 'Science', 'History']
scores = [95, 88, 92]

# dict() 생성자와 zip()을 결합하면 반복문 없이 한 줄로 매핑 가능
# 이는 내부적으로 C로 구현된 최적화 루프를 타게 되어 성능상 유리함
score_dict = dict(zip(subjects, scores))

print("--- 1.1 zip() 활용 결과 ---")
print(f"매핑된 딕셔너리: {score_dict}") 
# 결과: {'Math': 95, 'Science': 88, 'History': 92}
```


## 1.2 enumerate() 를 통한 인덱스 관리의 안정성 확보
리스트를 순회하면서 값과 해당 요소의 인덱스가 동시에 필요한 상황에서 range(len(nums))를 사용하는 것은 파이썬에서 지양해야할 안티 패턴 중 하나
- 장점 : 인덱스 변수를 수동으로 증가시키거나 계산할 필요가 없어 'Off-by-one Error' (경계 조건에서 1차이로 발생하는 오류) 를 근본적으로 차단한다


```
# [실전 예시] 특정 목표값의 모든 인덱스 위치 찾기
nums = [1,2,3,2,1]
target = 2

# enumerate(nums, start=0) 형식을 가지며 (index, value) 쌍을 순차적으로 반환
target_indices = [idx for idx, val in enumearte(nums) if val == target]

print("\n--- 1.2 enumerate() 활용 결과 ---")
print(f"타겟 '{target}'의 위치 리스트: {target_indices}") # 결과: [1, 3]
```


# 2. 문자열 연산과 조건 제어 최적화
## 2.1 join() 메서드와 불변 객체의 메모리 전략
파이썬에서 문자열 str 는 불변, Immutable 객체이다

반복문 내에서 + 연산자로 문자열을 계속 더하는 행위는 매 단계마다 새로운 문자열 객체를 메모리에 할당하고 기존 내용을 복사하므로, 단어 개수가 N개일 때 O(N^2) 의 심각한 성능 저하를 초래한다

- 해결책 : join 메서드는 결합할 모든 문자열의 크기를 미리 계산한 뒤 한 번의 메모리 할당으로 작업하므로 O(N) 의 선형 시간 내에 연산이 끝난다
- 그러니까 +로 계속 문자열을 끝에다 추가하면 그만큼 연산량이 제곱으로 증가하지만 join을 쓰면 한번에 다 처리하므로 문자열 개수만큼만 선형적으로 처리하는 것으로 보임

```
# [실전 예시] 리스트 내부 단어들을 문장으로 결합
words = ["Python", "is", "a", "powerful", "language"]

# 구분자(Separator)인 공백 " "을 기준으로 join 호출
# 이는 가독성뿐만 아니라 성능 면에서도 코딩 테스트의 필수 테크닉임
sentence = " ".join(words)

print("\n--- 2.1 join() 활용 결과 ---")
print(f"결합된 문장: {sentence}")
```

## 2.2 중첩 삼항 연산자의 가독성
복잡하지 않은 조건부 값 할당은 한 줄의 식으로 표현하여 코드의 밀도를 높일 수 있다

다만, 가독성을 해치지 않는 선에서 사용해야 한다

```
# [실전 예시] 수치 데이터의 삼항 조건 판별
n = 0

# 구조 : [참일 때의 값] if [조건] else [거짓일 때의 값]
# 아래와 같이 중첩하여 다중 분기(Positive / Zero / Negative) 처리 가능
status = "Positive" if n > 0 else ("Zero" if n == 0 else "Negative") 

print("\n--- 2.2 삼항 연산자 결과 ---")
print(f"숫자 {n}의 상태 판별: {status}")
```



# 3. 데이터 언패킹과 효율적인 멤버십 테스트
## 3.1 가변 언패킹 (*) 을 이용한 데이터 분리
리스트나 튜플에서 특정 요소만 변수로 할당하고 나머지는 별도의 리스트로 유지해야할 때, 슬라이싱보다 가변 언패킹(*) 연산자가 훨씬 직관적이다

```
# [실전 예시] 정렬된 점수 리스트에서 극단값 제외 및 중간 데이터 추출
# 데이터가 이미 정렬되어 있다고 가정 (예: [50, 70, 80, 90, 100])
scores = [50, 70, 80, 90, 100]

# 처음(최소) 과 끝(최대)을 추출하고 중간은 '*" 을 통해 리스트로 묶음
min_val, *mid_vals, max_val = scores

print("\n--- 3.1 가변 언패킹 결과 ---")
print(f"최소: {min_val}, 최대: {max_val}, 분석 대상: {mid_vals}")
```


## 3.2 in 연산자와 집합 (Set) 의 해시 메커니즘

특정 요소가 존재하는지 확인하는 멤버쉽 테스트에서 자료구조의 선택은 알고리즘의 성패를 가른다
- 리스트 (O(N)) : 순차 탐색을 수행하므로 데이터가 많아질수록 선형적으로 느려짐
- 집합 (O(1)) : 내부적으로 해시 테이블을 사용하여 데이터 크기와 무관하게 즉각적인 응답이 가능


```
# [실전 예시] 특정 문자의 포함 여부 판별
vowels = {'a', 'e', 'i', 'o', 'u'} # 리스트가 아닌 집합(Set)으로 선언
target_word = "apple"

# 집합에 대한 'in' 연산은 매우 빠름
contains_vowel = any(char in vowels for char in target_word)

print("\n--- 3.2 멤버십 테스트 결과 ---")
print(f"'{target_word}'에 모음 포함 여부: {contains_vowel}")
```


# 4. 함수형 프로그래밍과 커스텀 정렬
## 4.1 map() 과 lambda 를 통한 일괄 변환
리스트의 모든 요소에 동일한 함수를 적용할 때, 명시적인 for 루프 대신 map() 을 사용하면 코드가 간결해지고 의도가 명확해짐

```
# [실전 예시] 문자열 형태의 숫자 리스트를 정수 리스트로 변환
str_nums = ["1", "2", "3"]

# map(적용할 함수, 대상 Iterable)
# map 객체를 반환하므로 최종적으로 list() 변환이 필요함
int_nums = list(map(int, str_nums))

print("\n--- 4.1 map/lambda 결과 ---")
print(f"변환된 정수 리스트: {int_nums}")
```

## 4.2 슬라이싱의 Step 인자를 활용한 데이터 정제
슬라이싱 구문의 세 번째 인자인 step은 복잡한 반복 로직을 단 한 줄의 인덱싱으로 대체해줌

```
# [실전 예시] 특정 규칙(짝수 번 인덱스)에 따른 문자 추출
s = "abcdef"

# s[시작:끝:간력] -> 0번부터 끝까지 2칸 간격으로 추출
even_indexed_chars = s[::2]

print("\n--- 4.2 슬라이싱 Step 결과 ---")
print(f"짝수 인덱스 문자열: {even_indexed_chars}") # 결과: "ace"
```

## 4.3 key 인자와 lambda 를 이용한 다중 기준 정렬
파이썬의 정렬 알고리즘은 안정적이며, key 인자를 통해 복잡한 정렬 조건을 한 번에 부여할 수 있음

```
# [실전 예시] 길이순으로 정렬하되, 길이가 같으면 사전순 정렬
words = ["apple", "no", "banana", "cat"]

# key에 튜플을 반환하는 lambda 를 전달하면 (우선순위1, 우선수위2) 로 동작함
words.sort(key=lambda x: (len(x), x))

print("\n--- 4.3 커스텀 정렬 결과 ---")
print(f"정렬된 단어 리스트: {words}")
```


# 5. 집합 연산을 통한 데이터 비교 및 정제
두 집단의 데이터를 비교하거나 차이를 찾아야 할 때, 이중 반복문은 O(NxM) 의 비효율을 발생시킨다

집합의 차집합 연산을 사용하면 훨씬 효율적으로 해결할 수 있다

```
# [실전 예시] A 리스트에만 존재하는 고유 원소 추출
list_a = [1, 2, 2, 3]
list_b = [3, 4]

# set으로 변환 시 중복이 제거되며, '-' 연산자로 차집합 수행 가능
unique_to_a = list(set(list_a) - set(list_b))

print("\n--- 5. 집합 연산 결과 ---")
print(f"A에만 존재하는 원소(중복 제거): {unique_to_a}") # 결과: [1, 2]
```
